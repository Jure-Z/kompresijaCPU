cmake_minimum_required(VERSION 3.0...3.25)
project(
	LearnWebGPU # name of the project, which will also be the name of the visual studio solution if you use it
	VERSION 0.1.0 # any version number
	LANGUAGES CXX C # programming languages used by the project
)

# We specify that we want to create a target of type executable, called "App"
add_executable(App main.cpp "stb_image.h")

# Set some recommended properties to configure the version of C++ and treat warnings as errors
set_target_properties(App PROPERTIES
	CXX_STANDARD 17
	CXX_STANDARD_REQUIRED ON
	CXX_EXTENSIONS OFF
	COMPILE_WARNING_AS_ERROR ON
)

option(DEV_MODE "Set up development helper settings" ON)

if(DEV_MODE)
	target_compile_definitions(App PRIVATE
		INPUT_DIR="${CMAKE_CURRENT_SOURCE_DIR}/input"
		OUTPUT_DIR="${CMAKE_CURRENT_SOURCE_DIR}/output"
	)
else()
	target_compile_definitions(App PRIVATE
		INPUT_DIR="./input"
		OUTPUT_DIR="./output"
	)
endif()
